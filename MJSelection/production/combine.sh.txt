#!/bin/bash

selection=$1
sample=$2

if [[ ($sample = "All") || ($sample != "All" && $sample = "QCD") ]]; then rm ../baconbits/$selection/QCD.root;  hadd  ../baconbits/$selection/QCD.root   ../baconbits/$selection/*QCD*/*.root; fi

if [ $selection != "Pho" ]; then
    if [[ ($sample = "All") || ($sample != "All" && $sample = "DY") ]]; then rm ../baconbits/$selection/DY.root;    hadd  ../baconbits/$selection/DY.root     ../baconbits/$selection/*DYJets*_HT*/*.root; fi
    if [[ ($sample = "All") || ($sample != "All" && $sample = "DYNLO") ]]; then rm ../baconbits/$selection/DYNLO.root;  hadd  ../baconbits/$selection/DYNLO.root   ../baconbits/$selection/*DYJetsToLL_M-50_MINIAOD*/*.root; fi
    if [[ ($sample = "All") || ($sample != "All" && $sample = "Z") ]]; then rm ../baconbits/$selection/Z.root;    hadd  ../baconbits/$selection/Z.root     ../baconbits/$selection/*ZJetsToNuNu*/*.root; fi
    if [[ ($sample = "All") || ($sample != "All" && $sample = "ZNLO") ]]; then rm ../baconbits/$selection/ZNLO.root;    hadd  ../baconbits/$selection/ZNLO.root     ../baconbits/$selection/*DYJetsToNuNu*/*.root; fi
    if [[ ($sample = "All") || ($sample != "All" && $sample = "W") ]]; then rm ../baconbits/$selection/W.root;    hadd  ../baconbits/$selection/W.root     ../baconbits/$selection/*WJets*_HT*/*.root; fi
    if [[ ($sample = "All") || ($sample != "All" && $sample = "WNLO") ]]; then rm ../baconbits/$selection/WNLO.root;    hadd  ../baconbits/$selection/WNLO.root     ../baconbits/$selection/Spring15_a25ns_WJetsToLNu_amcatnlo_MINIAOD*/*.root; fi
    if [[ ($sample = "All") || ($sample != "All" && $sample = "T") ]]; then rm ../baconbits/$selection/T.root;    hadd  ../baconbits/$selection/T.root     ../baconbits/$selection/*_ST_*/*.root; fi
    if [[ ($sample = "All") || ($sample != "All" && $sample = "TT1L") ]]; then rm ../baconbits/$selection/TT1L.root;   hadd  ../baconbits/$selection/TT1L.root    ../baconbits/$selection/Spring15_a25ns_TTJets_amcatnlo_MINIAOD*mctt1l/*.root; fi
    if [[ ($sample = "All") || ($sample != "All" && $sample = "TT2L") ]]; then rm ../baconbits/$selection/TT2L.root;   hadd  ../baconbits/$selection/TT2L.root    ../baconbits/$selection/Spring15_a25ns_TTJets_amcatnlo_MINIAOD*mctt2l/*.root; fi
    if [[ ($sample = "All") || ($sample != "All" && $sample = "TTHAD") ]]; then rm ../baconbits/$selection/TTHAD.root;   hadd  ../baconbits/$selection/TTHAD.root    ../baconbits/$selection/Spring15_a25ns_TTJets_amcatnlo_MINIAOD*mctthad/*.root; fi
    if [[ ($sample = "All") || ($sample != "All" && $sample = "TTBST") ]]; then rm ../baconbits/$selection/TTBST.root;   hadd  ../baconbits/$selection/TTBST.root    ../baconbits/$selection/Spring15_a25ns_TTJets_amcatnlo_MINIAOD*mcbst/*.root; fi
    if [[ ($sample = "All") || ($sample != "All" && $sample = "TTCOM") ]]; then rm ../baconbits/$selection/TTCOM.root;   hadd  ../baconbits/$selection/TTCOM.root    ../baconbits/$selection/Spring15_a25ns_TTJets_amcatnlo_MINIAOD*mccom/*.root; fi
    if [[ ($sample = "All") || ($sample != "All" && $sample = "WW") ]]; then rm ../baconbits/$selection/WW.root;   hadd  ../baconbits/$selection/WW.root    ../baconbits/$selection/*WW*/*.root; fi
    if [[ ($sample = "All") || ($sample != "All" && $sample = "WZ") ]]; then rm ../baconbits/$selection/WZ.root;   hadd  ../baconbits/$selection/WZ.root    ../baconbits/$selection/*WZ*/*.root; fi
    if [[ ($sample = "All") || ($sample != "All" && $sample = "ZZ") ]]; then rm ../baconbits/$selection/ZZ.root;   hadd  ../baconbits/$selection/ZZ.root    ../baconbits/$selection/*ZZ*/*.root; fi

    if [[ ($selection = "Had") || ($selection = "Muo") || ($selection = "Zmm") ]]; then
	if [[ ($sample = "All") || ($sample != "All" && $sample = "Data") ]]; then rm ../baconbits/$selection/Data.root; hadd  ../baconbits/$selection/Data.root   ../baconbits/$selection/MET_*/*.root; fi
    fi

    if [[ ($selection = "Ele") || ($selection = "Zee") ]]; then
	if [[ ($sample = "All") || ($sample != "All" && $sample = "Data") ]]; then rm ../baconbits/$selection/Data.root; hadd  ../baconbits/$selection/Data.root   ../baconbits/$selection/SingleElectron_*/*.root; fi
    fi
fi

if [ $selection = "Pho" ]; then
    if [[ ($sample = "All") || ($sample != "All" && $sample = "G") ]]; then rm ../baconbits/$selection/G.root;    hadd  ../baconbits/$selection/G.root     ../baconbits/$selection/*GJets*_HT*/*.root; fi
    if [[ ($sample = "All") || ($sample != "All" && $sample = "Data") ]]; then rm ../baconbits/$selection/Data.root; hadd  ../baconbits/$selection/Data.root   ../baconbits/$selection/SinglePhoton_*/*.root; fi
fi
